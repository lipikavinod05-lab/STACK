bool backspaceCompare(char* s, char* t) {

    int c = 0;

    int tl = strlen(t);
    char s1[sl];
    char s2[tl];

    for (i = 0; i < sl; i++) {

            s1[++top] = s[i];
        } else if (s[i] == '#' && top >= 0) {
            top--;
        }
    }
    for (i = 0; i < tl; i++) {
        if (isalpha(t[i])) {
            s2[++top1] = t[i];
        } else if (t[i] == '#' && top1 >= 0) {
            top1--;
        }
    }
    if (top != top1)

    while (top >= 0) {

    }

}
