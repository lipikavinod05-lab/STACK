char* removeDuplicates(char* s) {
    int top = -1;
    int l = strlen(s);
    char *st = (char*)malloc((l + 1) * sizeof(char)); 
    
    for(int i = 0; i < l; i++) {
       
        if(top < 0 || st[top] != s[i]) {
            st[++top] = s[i];  
        }
       
        else if(st[top] == s[i]) {
            top--; 
        }
    }
    
    st[++top] = '\0';  
    return st;
}
