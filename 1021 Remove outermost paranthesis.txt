#include <stdio.h>
#include <stdlib.h>
#include <string.h>
char* removeOuterParentheses(char* s) {
    char st[100];int j=0;int key=0;int f=0;
    char *res=(char*)malloc(strlen(s)*sizeof(char));
    int top=-1;
    int i=0;
    for(i=0;i<strlen(s);i++)
    {   
        char k=s[i];
        if(k!=')')
        {
          st[++top]=k;  
        }
        else
        {
            top--;
        }
        if(top==-1 && i!=0)
        {   
            
            for(j=key+1;j<i;j++)
            {
                res[f++]=s[j];
            }
            key=i+1;
        } 
    
} res[f]='\0';
return res;}
